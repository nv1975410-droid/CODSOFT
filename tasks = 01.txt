tasks = []

while True:
    print("1. Add Task")
    print("2. View Tasks")
    print("3. Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        task = input("Enter task: ")
        tasks.append(task)
        print("Added")

    elif choice == "2":
        print("Tasks:")
        for task in tasks:
            print(task)

    elif choice == "3":
        print("Exit")
        break

    else:
        print("Wrong choice")
